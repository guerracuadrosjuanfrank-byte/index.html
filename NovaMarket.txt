<script>
let users=JSON.parse(localStorage.getItem('users'))||[];
let currentUser=JSON.parse(localStorage.getItem('currentUser'));
let products=JSON.parse(localStorage.getItem('products'))||[];
let cart=JSON.parse(localStorage.getItem('cart'))||[];
let chat=JSON.parse(localStorage.getItem('chat'))||[];

const cartCount=document.getElementById('cartCount');
const sellerPanel=document.getElementById('sellerPanel');
const productsDiv=document.getElementById('products');
const recommendDiv=document.getElementById('recommend');
const messages=document.getElementById('messages');

cartCount.innerText=cart.length;

if(currentUser && currentUser.role==='seller'){
 sellerPanel.classList.remove('hidden');
}

// LOGIN / REGISTER

document.getElementById('authForm').addEventListener('submit',e=>{
 e.preventDefault();
 const user={
  name:name.value,
  email:email.value,
  password:password.value,
  role:role.value
 };

 let existing=users.find(u=>u.email===user.email);

 if(!existing){
  users.push(user);
  localStorage.setItem('users',JSON.stringify(users));
  existing=user;
 }

 if(existing.password===user.password){
  currentUser=existing;
  localStorage.setItem('currentUser',JSON.stringify(existing));
  if(existing.role==='seller') sellerPanel.classList.remove('hidden');
 }
});

// PRODUCT RENDER

function render(list=products){
 productsDiv.innerHTML='';
 list.forEach((p,i)=>{
  const card=document.createElement('div');
  card.className='card';

  let media='';
  if(p.image) media+=`<img src="${p.image}">`;
  if(p.video) media+=`<video controls src="${p.video}"></video>`;

  card.innerHTML=`
  ${media}
  <h3>${p.name}</h3>
  <p>${p.description}</p>
  <p class="price">$${p.price}</p>
  <p class="small">Vendedor: ${p.seller}</p>
  <span class="tag">${p.shipping}</span>
  <button onclick="addToCart(${i})">Agregar al carrito</button>
  `;

  productsDiv.appendChild(card);
 });

 recommend();
}

// PRODUCT POST

document.getElementById('productForm').addEventListener('submit',e=>{
 e.preventDefault();

 const product={
  name:pname.value,
  description:description.value,
  price:price.value,
  image:image.value,
  video:video.value,
  shipping:shipping.value,
  seller:currentUser.name
 };

 products.push(product);
 localStorage.setItem('products',JSON.stringify(products));

 e.target.reset();
 render();
});

// SEARCH

function searchProducts(){
 const q=document.getElementById('search').value.toLowerCase();
 const filtered=products.filter(p=>p.name.toLowerCase().includes(q));
 render(filtered);
}

// CART

function addToCart(i){
 cart.push(products[i]);
 localStorage.setItem('cart',JSON.stringify(cart));
 cartCount.innerText=cart.length;
}

function checkout(){
 if(cart.length===0){alert('Carrito vacío');return;}
 const payment=prompt('Pago: tarjeta / transferencia');
 alert('Pago procesado con '+payment);
 cart=[];
 localStorage.setItem('cart',JSON.stringify(cart));
 cartCount.innerText=0;
}

// AI RECOMMENDATION (simple)

function recommend(){
 recommendDiv.innerHTML='';
 let random=[...products].sort(()=>0.5-Math.random()).slice(0,3);
 random.forEach(p=>{
  const card=document.createElement('div');
  card.className='card';
  card.innerHTML=`<h4>${p.name}</h4><p>$${p.price}</p>`;
  recommendDiv.appendChild(card);
 });
}

// CHAT

function renderChat(){
 messages.innerHTML='';
 chat.forEach(m=>messages.innerHTML+=`<div>${m}</div>`);
}

function sendMessage(){
 const input=document.getElementById('chatInput');
 if(!input.value)return;
 chat.push(input.value);
 localStorage.setItem('chat',JSON.stringify(chat));
 input.value='';
 renderChat();
}

render();
renderChat();
</script>
