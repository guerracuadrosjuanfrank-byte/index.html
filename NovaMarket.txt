<script>

// CONFIGURACIÓN FIREBASE
const firebaseConfig = {
apiKey:"TU_API_KEY",
authDomain:"TU_DOMINIO",
projectId:"TU_PROJECT_ID",
storageBucket:"TU_BUCKET",
messagingSenderId:"TU_ID",
appId:"TU_APP_ID"
};

firebase.initializeApp(firebaseConfig);

const db = firebase.firestore();
const auth = firebase.auth();

let carrito=[];
let total=0;

// REGISTRO
function registrar(){

let email=document.getElementById("email").value;
let password=document.getElementById("password").value;

auth.createUserWithEmailAndPassword(email,password);

}

// LOGIN
function login(){

let email=document.getElementById("email").value;
let password=document.getElementById("password").value;

auth.signInWithEmailAndPassword(email,password);

}

// PUBLICAR PRODUCTO
function publicar(){

let nombre=document.getElementById("nombre").value;
let precio=parseFloat(document.getElementById("precio").value);
let imagen=document.getElementById("imagen").value;
let video=document.getElementById("video").value;

db.collection("productos").add({

nombre:nombre,
precio:precio,
imagen:imagen,
video:video

});

}

// MOSTRAR PRODUCTOS
db.collection("productos").onSnapshot((snapshot)=>{

let html="";

snapshot.forEach((doc)=>{

let p=doc.data();

html+=`
<div class="producto">

<h3>${p.nombre}</h3>

<img src="${p.imagen}" width="150">

<p>$${p.precio}</p>

${p.video ? `<iframe width="200" src="${p.video}"></iframe>`:""}

<br>

<button onclick='agregar(${p.precio})'>Agregar al carrito</button>

</div>
`;

});

document.getElementById("productos").innerHTML=html;

});

// CARRITO
function agregar(precio){

carrito.push(precio);

total+=precio;

document.getElementById("total").innerText=total;

}

// COMPRAR
function comprar(){

let comision=total*0.10;

db.collection("pedidos").add({

total:total,
comision:comision

});

alert("Compra realizada");

carrito=[];
total=0;

document.getElementById("total").innerText=0;

}

// PANEL
db.collection("pedidos").onSnapshot((snapshot)=>{

let ventas=snapshot.size;

let dinero=0;

snapshot.forEach((doc)=>{

dinero+=doc.data().comision;

});

document.getElementById("ventas").innerText=ventas;
document.getElementById("ganancia").innerText="$"+dinero;

});

// BUSCADOR
function buscarProducto(){

let filtro=document.getElementById("buscar").value.toLowerCase();

let productos=document.getElementsByClassName("producto");

for(let p of productos){

let texto=p.innerText.toLowerCase();

p.style.display=texto.includes(filtro)?"block":"none";

}

}
